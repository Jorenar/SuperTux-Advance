50 uint16_t TileManager::create_vram_tile(uint16_t rom_id)
51 {
52 #ifdef ENABLE_DYNAMIC_TILE_ALLOCATION
53  if (rom_id == 0) {
54  // Transparent tile is a special case, ignore
55  return 0;
56  }
57  else if (rom2vram[rom_id] == NO_TILE) {
58  // tile not allocated, so allocate it
59  if (next_free == 0) {
60  console << "Out of Tilespace for: " << rom_id << "\n";
61  return 0;
62  }
63  else {
64  uint16_t vram_id = next_free;
65 
66  rom2vram[rom_id] = vram_id;
67  vram2rom[vram_id] = rom_id;
68 
69  next_free = freelist[vram_id];
70 
71  refcount[vram_id] = 1;
72 
73  upload_tile(rom_id, vram_id);
74 
75  return rom2vram[rom_id];
76  }
77  }
78  else {
79  // tile already allocated, so increase the refcount
80  refcount[rom_id] += 1;
81  return rom2vram[rom_id];
82  }
83 #else
84  // Simple 1:1 mapping from rom to ram
85  return rom_id % MAX_VRAM_ID;
86 #endif
87 }
