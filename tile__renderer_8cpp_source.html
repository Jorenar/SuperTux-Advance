42 void TileRenderer::set_tilemap_offset(uint8_t layer_num, int16_t x_offset, int16_t y_offset)
43 {
44  Layer& layer = layers[layer_num];
45 
46  // Cut new offset to the limits of the layer
47  // FIXME: Might be useful to make limits configurable per layer
48  x_offset = Math::mid(0, (int)x_offset, layer.tilemap->get_width() *8 - 32*8);
49  y_offset = Math::mid(0, (int)y_offset, layer.tilemap->get_height()*8 - 20*8);
50 
51  //copy_tilemap(layer_num);
52 
53  int x_tile = layer.x_offset/8;
54  int y_tile = layer.y_offset/8;
55 
56  int new_x_tile = x_offset/8;
57  int new_y_tile = y_offset/8;
58 
59  if (x_tile != new_x_tile ||
60  y_tile != new_y_tile) {
61  int start_x = Math::min(x_tile, new_x_tile);
62  int start_y = Math::min(y_tile, new_y_tile);
63 
64  int end_x = Math::max(x_tile, new_x_tile);
65  int end_y = Math::max(y_tile, new_y_tile);
66 
67  int left = start_x;
68  int top = start_y;
69 
70  if (new_x_tile > x_tile) {
71  start_x += 32;
72  end_x += 32;
73  }
74 
75  if (new_y_tile > y_tile) {
76  start_y += 32;
77  end_y += 32;
78  }
79 
80  // Version refresh
81  for (int y = start_y; y < end_y && (y < layer.tilemap->get_height()); ++y) {
82  for (int x = left; x < left+32; ++x) {
83  uint16_t& tile = layer.vram[(y%32) * 32 + (x%32)];
84 
85  tile_manager->delete_vram_tile(tile);
86  tile = tile_manager->create_vram_tile(layer.tilemap->get_data()[y * layer.tilemap->get_width() + x]);
87  }
88  }
89 
90  // Horizontal refresh
91  for (int y = top; y < top+32 && (y < layer.tilemap->get_height()); ++y) {
92  for (int x = start_x; x < end_x; ++x) {
93  uint16_t& tile = layer.vram[(y%32) * 32 + (x%32)];
94 
95  tile_manager->delete_vram_tile(tile);
96  tile = tile_manager->create_vram_tile(layer.tilemap->get_data()[y * layer.tilemap->get_width() + x]);
97  }
98  }
99 
100  // FIXME: The regions of vertical and horizontal refresh may overlap, could be optimized further
101 
102  // FIXME: Some off-by-one bug seems to be hiding here, diagonal scrolling causes slight trouble
103  }
104 
105  layer.x_offset = x_offset;
106  layer.y_offset = y_offset;
107 
108  bg_scroll scroll;
109  scroll.x = layer.x_offset;
110  scroll.y = layer.y_offset;

120 void TileRenderer::copy_tilemap(uint8_t layer_num)
121 {
122  Layer& layer = layers[layer_num];
123 
124  uint16_t start_x = layer.x_offset/8;
125  uint16_t start_y = layer.y_offset/8;
126  for (uint8_t x = start_x; x < start_x + 32; ++x) {
127  for (uint8_t y = start_y; y < start_y + 32 && (y < layer.tilemap->get_height()); ++y) {
128  uint16_t& tile = layer.vram[(y%32) * 32 + (x%32)];
129  tile_manager->delete_vram_tile(tile);
130  tile = tile_manager->create_vram_tile(layer.tilemap->get_data()[(y) * layer.tilemap->get_width() + (x)]);
131  }
132  }
133 }
